import PyPDF2
from PyPDF2 import PdfReader
from PyPDF2 import PdfWriter as w
import os
import glob
listing = glob.glob(os.getcwd()+'/[0-9][0-9]-[0-9][0-9] [ЭС]-[ЭС].pdf')
dict=dict(list())
for ticketFile in listing:
	r=PdfReader(ticketFile)
	p=r.pages[0]
	str=p.extract_text()
	date=str[str.find("ДАТАПРОЕЗДА ")+12:str.find("ДАТАПРОЕЗДА ")+22]
	if date in dict:
		dict[date] = [dict[date], ticketFile]
	else:
		dict[date] = ticketFile
print(dict)
